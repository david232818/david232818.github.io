unsigned int ror_8(unsigned int x, unsigned int n)
{
        unsigned int tmp;

        x &= SETBIT(8);
        tmp = x & SETBIT(n);
        x >>= n;
        tmp <<= (8 - n);
        x |= tmp;
        return x;
}
